Capítulo 09

Cinco letras, cinco decisões de código

S

Single Responsibility

Uma classe, uma regra de item.

O

Open/Closed

Item novo = classe nova + registro, zero edição.

L

Liskov Substitution

Qualquer estratégia substitui a base sem quebrar nada.

I

Interface Segregation

Um método só: update(item).

D

Dependency Inversion

GildedRose depende da abstração UpdaterRegistry, injetada no construtor — você vai ver o código exato disso já nos próximos slides.

class BackstagePassesUpdater(ItemUpdater):
    """Backstage passes.

    Quality sobe conforme o show se aproxima
    e despenca a zero apos o show acontecer.

    Regras de incremento por dia:
        * +1 sempre.
        * +1 extra quando sell_in < 11.
        * +1 extra quando sell_in < 6.
    """

    def update(self, item: Item) -> None:
        item.quality = min(MAX_QUALITY, item.quality + 1)
        if item.sell_in < 11:
            item.quality = min(MAX_QUALITY, item.quality + 1)
        if item.sell_in < 6:
            item.quality = min(MAX_QUALITY, item.quality + 1)
        item.sell_in -= 1
        if item.sell_in < 0:
            item.quality = 0

class UpdaterRegistry:
    """Mapeia nome de item para ItemUpdater.

    A resolucao tem fallback para correspondencia
    por predicado e, por fim, para um updater default.
    """

    def __init__(self, default_updater: ItemUpdater) -> None:
        """Inicializa um registry vazio com um updater default.

        Args:
            default_updater: Estrategia usada quando nenhum
                nome ou predicado registrado corresponde ao item.
        """
        self._default_updater: ItemUpdater = default_updater
        self._updaters_by_name: Dict[str, ItemUpdater] = {}
        self._predicate_updaters: List[Tuple[NamePredicate, ItemUpdater]] = []

    def register(self, item_name: str, updater: ItemUpdater) -> "UpdaterRegistry":
        """Associa um nome exato de item a um updater.

        Args:
            item_name: Nome exato do item (ex. "Aged Brie").
            updater: Estrategia a ser usada para esse item.

        Returns:
            A propria instancia, para permitir encadeamento.
        """
        self._updaters_by_name[item_name] = updater
        return self

Como o registry resolve a estratégia certa

depois · updaters/registry.py (continuação) 
def resolve(self, item_name: str) -> ItemUpdater:
    """Resolve a estrategia adequada
    para um nome de item.

    Args:
        item_name: Nome do item a resolver.

    Returns:
        O ItemUpdater correspondente,
        ou o updater default.
    """
    if item_name in self._updaters_by_name:
        return self._updaters_by_name[item_name]
    for predicate, updater in self._predicate_updaters:
        if predicate(item_name):
            return updater
    return self._default_updater
def build_default_registry() -> UpdaterRegistry:
    """Constroi o registry padrao da aplicacao.

    Returns:
        Registry pronto para uso, podendo ser
        estendido antes de injetar em GildedRose.
    """
    registry = UpdaterRegistry(
        default_updater=NormalItemUpdater()
    )
    registry.register(AGED_BRIE, AgedBrieUpdater())
    registry.register(SULFURAS, SulfurasUpdater())
    registry.register(
        BACKSTAGE_PASSES, BackstagePassesUpdater()
    )
    registry.register_by_predicate(
        is_conjured, ConjuredItemUpdater()
    )
    return registry

class GildedRose(object):
    """Inventario da loja.

    Depende apenas da abstracao UpdaterRegistry
    (DIP) -- nunca decide a regra de nenhum item.
    """

    def __init__(
        self,
        items: List[Item],
        updater_registry: Optional[UpdaterRegistry] = None,
    ) -> None:
        """Inicializa o inventario.

        Args:
            items: Itens do inventario.
            updater_registry: Registry de estrategias.
                Se None, usa o registry default.
        """
        self.items: List[Item] = items
        self._updater_registry = (
            updater_registry or build_default_registry()
        )

    def update_quality(self) -> None:
        """Avanca um dia para todos os itens."""
        for item in self.items:
            updater = self._updater_registry.resolve(item.name)
            updater.update(item)
